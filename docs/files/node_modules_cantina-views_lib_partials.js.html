/**
 * Cantina Views - Partials Plugin
 * --------------------------------
 *
 * Exposes an api for registering views partials.
 *
 * @module cantina
 * @submodule views
 */

// Module dependencies.
var diveSync = require('diveSync'),
    async = require('async'),
    cons = require('consolidate');

/**
 * Views Partials Plugin
 * ---------------------
 *
 * Exposes `app.views.partials`.
 *
 * Loaded automatically by the views plugin.
 *
 * @class views.plugin.partials
 */
var partials = module.exports = {
  name: 'views-partials',
  description: 'Register view partials'
};

/**
 * Attach the partials plugin to an application.
 *
 * @method attach
 * @param options {Object} Plugin options.
 * @for views.plugin.partials
 */
partials.attach = function(options) {
  var app = this;

  options = options || {};
  app.utils.defaults(options, app.views.options);

  /**
   * Partials registry.
   *
   * @property _partials
   * @private
   * @type {Object}
   * @for app.views
   */
  app.views._partials = app.views._partials || {};

  /**
   * Register partials.
   *
   * @method partials
   * @param [match] {String|RegExp} A string or regex to limit partial rendering
   *   to matching urls.
   * @param source {String} The path to a directory of views.  Registered
   *   namespaces will be honored. The directory will be recursively searched
   *   for views matching options.ext,  and those will be rendered and attached
   *   to the template data as properties matching the filename with the
   *   extension stripped.
   * @for app.views
   */
  app.views.partials = function(match, source) {
    var dir, view, parts, assign;

    if (arguments.length === 1) {
      source = match;
      match = /.*/;
    }

    match = app.views._stringifyPath(match);
    app.views._partials[match] = app.views._partials[match] || {};

    // Get the full path to the source (apply registered namespace).
    if (dir = app.views.findDir(source)) {
      // 'Dive' through the partials directory and register all the templates.
      diveSync(dir, function(err, file) {
        if (err) throw err;
        if (file.indexOf('.' + options.ext) > 0) {
          view = file.replace('.' + options.ext, '');
          parts = view.replace(new RegExp(dir + '\/?'), '').split('/');
          if (parts.length) {
            assign = app.views._partials[match];
            for (var i = 0, len = parts.length; i < len; i++) {
              if (i < len - 1) {
                if (!assign[parts[i]]) {
                  assign[parts[i]] = {};
                }
                assign = assign[parts[i]];
              }
              else {
                assign[parts[i]] = view;
              }
            }
          }
        }
      });
    }
  }

  /**
   * Process the app partials, rendering them and merging them with the passed
   * templateData object.
   *
   * Should be invoked in the router scope.
   *
   * @method _processPartials
   * @private
   * @param templateData {Object} The template data object to extend.
   * @param callback {Function} A callback to once the partials have been
   *   processed.
   * @for app.views
   */
  app.views._processPartials = function(templateData, callback) {
    var self = this,
        tasks = [],
        reqPath = app.utils.parseUrl(self.req.url).pathname;

    // Merge default options with template data.
    app.utils.defaults(templateData, options);

    // Check for cached partials for this requests so we don't run
    // them more than once.
    if (self.req._partialsData) {
      app.utils.defaults(templateData, self.req._partialsData);
      return callback(null);
    }

    // Loop through partials and render the ones that match the req url.
    for (var match in app.views._partials) {
      if (app.views._partials.hasOwnProperty(match)) {
        (function(match, partials) {
          if (reqPath.match(new RegExp(match))) {
            tasks.push(function(done) {
              app.views._renderPartials(partials, templateData, done);
            });
          }
        })(match, app.views._partials[match]);
      }
    }
    if (tasks.length) {
      async.parallel(tasks, function(err, results) {
        if (err) throw err;
        self.req._partialsData = {};
        for (var i = 0, len = results.length; i < len; i++) {
          app.utils.defaults(self.req._partialsData, results[i]);
        }
        app.utils.defaults(templateData, self.req._partialsData);
        return callback(err);
      });
    }
    else {
      return callback(null);
    }
  }

  /**
   * Recursively loop through partials and render them.
   *
   * @method _renderPartials
   * @private
   * @param  partials {Object} The top-level partials to render.
   * @param  templateData {Object} The template data object to extend.
   * @param  callback {Function} A callback to invoke after all the partials
   *   have been rendered and merged into the template data.
   * @for app.views
   */
  app.views._renderPartials = function(partials, templateData, callback) {
    var tasks = {};

    for (var key in partials) {
      if (partials.hasOwnProperty(key)) {
        (function(key, partial) {
          var template = partial + '.' + options.ext;
          if (typeof partial === 'object') {
            // Recurse
            tasks[key] = function(done) {
              app.views._renderPartials(partial, templateData, done);
            };
          }
          else {
            // Render
            tasks[key] = function(done) {
              cons[options.engine](template, templateData, done);
            };
          }
        })(key, partials[key]);
      }
    }

    async.parallel(tasks, function(err, results) {
      callback(null, results);
    })
  }
}
