/**
 * Cantina Views
 * -------------
 *
 * A full-featured views system for cantina apps.  Provides a views registry,
 * partials registry, and support for 14+ templating systems
 * (via consolidatejs).  Allows route handlers to render views via
 * `this.render`.
 *
 * @module cantina
 * @submodule views
 * @main views
 */

// Modules dependencies.
var utils = require('cantina').plugins.utils;

// Expose this service's package info.
require('pkginfo')(module);

/**
 * Views
 * -----
 *
 * Cantina views exposes a plugin for use in your application.  You can
 * use the plugin like:
 *
 *     app.use(require('cantina-views').plugin, {
 *       // options
 *     });
 *
 * See the `cantina-views.plugin` docs for more info.
 *
 * @class views
 */
var views = exports = module.exports;

// Lazy-load sub-modules.
utils.lazy(views, __dirname, {
  plugin: './plugin'
