/**
 * Cantina Views - Helpers Plugin
 * --------------------------------
 *
 * Exposes an api for adding dynamic helpers for binding template data.
 *
 * @module cantina
 * @submodule views
 */

// Module dependencies.
var async = require('async');

/**
 * Views Helpers Plugin
 * --------------------
 *
 * Exposes `app.views.helper` and `app.views.clearHelpers`.
 *
 * Loaded automatically by the views plugin.
 *
 * @class views.plugin.helpers
 */
var helpers = module.exports = {
  name: 'views-helpers',
  description: 'Bind static and dynamic template data for views'
};

/**
 * Attach the helpers plugin to an application.
 *
 * @method attach
 * @param options {Object} Plugin options.
 * @for views.plugin.helpers
 */
helpers.attach = function(options) {
  var app = this;

  /**
   * Helpers registry.
   *
   * @property _helpers
   * @private
   * @type {Object}
   * @for app.views
   */
  app.views._helpers = {};

  /**
   * Register a views helper.
   *
   * A helper can either be an object literal that will be merged with the
   * other template data, or a function. Function ('dynamic') helpers allow
   * you to add template data based on a request.
   *
   * Helper functions will be called in the router scope (this.req, this.res,
   * this.app) during the view rendering phase.  Helper funtions accept a
   * callback that should be called with the arguments `callback(err, data)`.
   *
   * @method helper
   * @param [match] {String|RegExp} If specified, the helper will only be
   *   applied for urls that match this pattern.
   * @param helper {Object|Function} Template data or a helper function.
   * @for app.views
   */
  app.views.helper = function(match, helper) {
    if (arguments.length === 1) {
      helper = match;
      match = /.*/;
    }

    match = app.views._stringifyPath(match);

    // Instantiate path array.
    if (!app.views._helpers[match]) {
      app.views._helpers[match] = [];
    }

    app.views._helpers[match].push(helper);
  };

  /**
   * Clear all views helpers.
   *
   * @method clearHelpers
   * @param [match] {String|RegExp} Only clear helpers registerd under this
   *   match pattern.
   * @for app.views
   */
  app.views.clearHelpers = function(match) {
    if (match) {
      match = app.views._stringifyPath(match);
      if (app.views._helpers[match]) {
        delete app.views._helpers[match];
      }
    }
    else {
      app.views._helpers = {};
    }
  }

  /**
   * Process views helpers.  Shoud be invoked in the 'router' scope.
   *
   * @method _processHelpers
   * @private
   * @param  templateData {Object} Template data to be extended.
   * @param  callback {Function} (err) Callback to be invoked after all registered
   *   helpers have been processed.
   * @for app.views
   */
  app.views._processHelpers = function(templateData, callback) {
    var self = this,
        tasks = [],
        reqPath = app.utils.parseUrl(self.req.url).pathname,
        helpersData = {};

    // Check for cached helpers data for this requests so we don't run
    // them more than once.
    if (self.req.helpersData) {
      app.utils.defaults(templateData, self.req.helpersData);
      return callback(null);
    }

    for (var match in app.views._helpers) {
      if (app.views._helpers.hasOwnProperty(match)) {
        (function(match, helpers) {
          if (reqPath.match(new RegExp(match))) {
            for (var i = 0, len = helpers.length; i < len; i++) {
              (function(helper) {
                if (typeof helper === 'function') {
                  tasks.push(function(done) {
                    helper.call(self, function(err, data) {
                      if (data) {
                        app.utils.defaults(helpersData, data);
                      }
                      done(err);
                    });
                  });
                }
                else {
                  app.utils.defaults(helpersData, helper);
                }
              })(helpers[i]);
            }
          }
        })(match, app.views._helpers[match]);
      }
    }
    if (tasks.length) {
      async.parallel(tasks, function(err, results) {
        app.utils.defaults(templateData, helpersData);
        self.req.helpersData = helpersData;
        return callback(err);
      });
    }
    else {
      app.utils.defaults(templateData, helpersData);
      self.req.helpersData = helpersData;
      return callback(null);
    }
  };
}
